function validateEmailDomain(email) {
  console.log("validating e-mail: " + email);
  remoteCallback("email-domain-errors", [email], function(r) {
    console.log("validation response: " + r);
    if (r) {
      prepend(byId("loginForm"), html(["div.help.email", "wrong email domain"]));
      return false;
    }
    true;
  })
}

(defn validate-email-domain [email]
  (. js/console log (str "validating e-mail: " email))
  (remote-callback :email-domain-errors [email] (fn [r]
    (. js/console log (str "validation response: " r))
    (if r
      (do (prepend! (by-id "loginForm") (html [:div.help.email "wrong email domain"]))
          false)
      true))))
